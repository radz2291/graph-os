## 💾 Data Persistence Patterns
### Local Storage Pattern (Browser-Friendly)
```typescript
// Storage Manager - No backend needed
export class StorageManager {
private key: string;
constructor(key: string) {
this.key = key;
}
// Load from localStorage
load(): T[] {
try {
const data = localStorage.getItem(this.key);
return data ? JSON.parse(data) : [];
} catch (err) {
console.error('Failed to load from localStorage:', err);
return [];
}
}
// Save to localStorage
save(items: T[]): void {
try {
localStorage.setItem(this.key, JSON.stringify(items));
} catch (err) {
console.error('Failed to save to localStorage:', err);
}
}
// Add item
add(item: T): void {
const current = this.load();
this.save([...current, item]);
}
// Update item
update(id: string, updates: Partial): void {
const current = this.load();
this.save(current.map(item =>
item.id === id ? { ...item, ...updates } : item
));
}
// Delete item
delete(id: string): void {
const current = this.load();
this.save(current.filter(item => item.id !== id));
}
}
// Usage
const taskStorage = new StorageManager('tasks');
function TasksView() {
const [tasks, setTasks] = useState([]);
useEffect(() => {
setTasks(taskStorage.load());
}, []);
const handleCreateTask = (task: Task) => {
taskStorage.add(task);
setTasks(taskStorage.load());
emitSignal('TASK.CREATE', task);
};
return ;
}
